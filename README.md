# react-tab

### 安装

```shell
npm i @vmo-fed/video-player -S
```

### 使用

```jsx
import VideoPlayer  from 'video-player';

class App extends React.Component {
  render() {
    return (
      <VideoPlayer src="http://www.mediacollege.com/video-gallery/testclips/20051210-w50s_56K.flv" />
    );
  }
}
```
